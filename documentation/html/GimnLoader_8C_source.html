264 
265  // função para debug

267  std::vector<int> aux;
268 
269  if (!port.empty())
270  this->port.push_back(stoi(port));
271  if (!chipID.empty())
272  this->chipID.push_back(stoi(chipID));
273  if (!inCoincidence.empty())
274  {
275  separatedCoincidences = splitString(inCoincidence, ',');
276  for (auto &inCoin : separatedCoincidences)
277  {

287

292  if (!crystalType.empty())
